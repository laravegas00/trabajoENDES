-- Consulta con WHERE y una condición

SELECT ID, n_chasis
FROM Pilotos
WHERE nacionalidad = 'España'
ORDER BY n_chasis;

-- Consulta con WHERE y operadores logicos

SELECT ID
FROM MECANICOS
WHERE ESPECIALIDAD = 'Motor' AND AÑOS_EXPERIENCIA = 5
ORDER BY ID;

-- Consulta con BETWEEN

SELECT *
FROM COCHES
WHERE potencia_motor BETWEEN 800 AND 900
ORDER BY km;

-- Consulta con NOT IN

SELECT *
FROM CARRERAS
WHERE PAIS NOT IN 'España'
ORDER BY CODIGO_CARRERA;

-- Consulta usando LIKE con comodines (%, _)

SELECT NOMBRE, APE1, APE2
FROM TRABAJADORES
WHERE (APE1 LIKE '%ez%' AND APE2 LIKE '%ez%')
ORDER BY NOMBRE;

-- Consulta usando operadores aritmeticos

SELECT ID, ESPECIALIDAD, AÑOS_EXPERIENCIA, reparacionesrealizadas 
FROM MECANICOS
WHERE REPARACIONESREALIZADAS / AÑOS_EXPERIENCIA >= 1
ORDER BY ID;

-- Consulta usando MIN

SELECT MIN(HORAS_EXPERIENCIA)
FROM ENTRENADORES
WHERE HORAS_EXPERIENCIA > 500
ORDER BY ID;

-- Consulta usando MAX

SELECT MAX(POTENCIA_MOTOR)
FROM COCHES
WHERE AÑO_MODELO_COCHE = 2023
ORDER BY POTENCIA_MOTOR;

-- Consulta usando SUM 

SELECT SUM(KM) AS km_recorridos
FROM COCHES
WHERE AÑO_MODELO_COCHE = 2023
ORDER BY km_recorridos;

-- Consulta usando AVG

SELECT AVG(DISTANCIA_CIRCUITO)
FROM CARRERAS
WHERE RESULTADO LIKE 'Victoria%'
ORDER BY CODIGO_CARRERA;

-- Consulta usando COUNT

SELECT COUNT(TIPO)
FROM TRABAJADORES
WHERE TIPO = 'Pilot'
ORDER BY TIPO;

-- Consulta con NOT IN

SELECT *
FROM CARRERAS
WHERE PAIS NOT IN 'España'
ORDER BY CODIGO_CARRERA;

-- Consulta usando LIKE con comodines (%, _)

SELECT NOMBRE, APE1, APE2
FROM TRABAJADORES
WHERE (APE1 LIKE '%ez%' AND APE2 LIKE '%ez%')
ORDER BY NOMBRE;
